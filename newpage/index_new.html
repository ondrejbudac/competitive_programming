using namespace std;


class BishopMove {
public:
	int howManyMoves(int r1, int c1, int r2, int c2) {
		if ((abs(r1-r2) % 2) != (abs(c1-c2) % 2)) return -1;
		if ((r1==r2) && (c1 == c2)) return 0;
		if (abs(r1-r2) == abs(c1-c2)) return 1;
		return 2;
	}
};
